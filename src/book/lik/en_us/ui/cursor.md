### Cursor

```lua
Cursor()
    :uiKit(kit)
    :sizeRate(20)
    :textureAim({ normal = "aim\\white", ally = "aim\\green", enemy = "aim\\red", neutral = "aim\\gold" })
    :textureDrag({ alpha = 170, normal = "drag\\normal" })
    :textureSquare({ alpha = 130, allow = "square\\white", ban = "square\\red" })
```
